<script>
// ---------- Language ----------
let lang = localStorage.getItem("lang") || "ar";

const text = {
  ar: { days: ["أحد","إثن","ثلا","أرب","خم","جم","سب"] },
  en: { days: ["Sun","Mon","Tue","Wed","Thu","Fri","Sat"] }
};

function changeLang(l){
  lang = l;
  localStorage.setItem("lang", l);
  renderAll();
}

// ---------- Calendars ----------
let calendars = JSON.parse(localStorage.getItem("calendars")) || [];

function addCalendar(){
  const name = document.getElementById("calendarName").value.trim();
  if(!name) return;
  calendars.push({ name, days:{} });
  save();
  renderAll();
  document.getElementById("calendarName").value="";
}

function save(){
  localStorage.setItem("calendars", JSON.stringify(calendars));
}

// ---------- Hijri ----------
function getHijri(date){
  return new Intl.DateTimeFormat('ar-TN-u-ca-islamic', { day:'numeric', month:'numeric' }).format(date);
}

// ---------- Render ----------
function renderAll(){
  const container = document.getElementById("calendars");
  container.innerHTML = "";

  calendars.forEach((cal, cIndex)=>{
    const div = document.createElement("div");
    div.className = "calendar";

    div.innerHTML = `<div class="calendar-title">${cal.name}</div>`;

    const grid = document.createElement("div");
    grid.className = "grid";

    text[lang].days.forEach(d=>{
      const h = document.createElement("div");
      h.style.fontWeight="bold";
      h.style.textAlign="center";
      h.innerText=d;
      grid.appendChild(h);
    });

    const year = new Date().getFullYear();
    const start = new Date(year,0,1);
    const end = new Date(year,11,31);

    for(let i=0;i<start.getDay();i++){
      grid.appendChild(document.createElement("div"));
    }

    for(let d=start; d<=end; d.setDate(d.getDate()+1)){
      const dateKey = d.toISOString().split("T")[0];
      const cell = document.createElement("div");
      cell.className = "day";
      cell.innerHTML = `<div>${d.getDate()}</div><div>${getHijri(new Date(d))}</div>`;

      if(cal.days[dateKey]?.checked) cell.classList.add("checked");
      if(cal.days[dateKey]?.special) cell.classList.add("special");

      const today = new Date().toISOString().split("T")[0];
      if(dateKey === today) cell.classList.add("today");

      cell.onclick = ()=>{
        cal.days[dateKey] = cal.days[dateKey] || {};
        cal.days[dateKey].checked = !cal.days[dateKey].checked;
        save(); renderAll();
      };

      cell.oncontextmenu = (e)=>{
        e.preventDefault();
        cal.days[dateKey] = cal.days[dateKey] || {};
        cal.days[dateKey].special = !cal.days[dateKey].special;
        save(); renderAll();
      };

      grid.appendChild(cell);
    }

    div.appendChild(grid);
    container.appendChild(div);
  });
}

renderAll();
</script>
